#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aColor;

uniform vec3 bounds;

uniform vec3 playerPos;

out vec3 color;


void main()
{
    vec3 posPlayerN = vec3((aPos.x - playerPos.x) / bounds.x, (aPos.y - playerPos.y) / bounds.y, (aPos.z - playerPos.z) / bounds.z);

    gl_Position = vec4(posPlayerN.x, posPlayerN.y, posPlayerN.z*posPlayerN.z, posPlayerN.z);
    color = aColor;
}
